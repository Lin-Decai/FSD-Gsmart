//ROI+RANSAC+欧几里德+形态学
#include "rclcpp/rclcpp.hpp"
#include "sensor_msgs/msg/point_cloud2.hpp"
#include "pcl_conversions/pcl_conversions.h"
#include "pcl/point_types.h"
#include "pcl/filters/passthrough.h"
#include "pcl/filters/voxel_grid.h"
#include "pcl/filters/extract_indices.h"
#include "pcl/segmentation/sac_segmentation.h"
#include "pcl/segmentation/extract_clusters.h"
#include "pcl/common/centroid.h"
#include "geometry_msgs/msg/pose_array.hpp"
#include <utility> 
#include <pcl/common/geometry.h>

class HesaiProcessor : public rclcpp::Node
{
public:
    HesaiProcessor() : Node("hesai_processor")
    {
        sub_lidar = this->create_subscription<sensor_msgs::msg::PointCloud2>(
            "/lidar_points",
            rclcpp::SensorDataQoS().keep_last(5),//使用适合传感器数据的默认 QoS 配置，保留最近 5 个消息
            [this](const sensor_msgs::msg::PointCloud2::ConstSharedPtr msg) {
                this->pointCloudCallback(msg);
            });//使用 Lambda 表达式将接收到的消息传递给成员函数pointCloudCallback
//创建四个发布者，分别发布 ROI、地面滤波、聚类结果、锥体位置
        roi_pub_ = this->create_publisher<sensor_msgs::msg::PointCloud2>("/hesai/roi_cloud", 10);
        ground_pub_ = this->create_publisher<sensor_msgs::msg::PointCloud2>("/hesai/ground_filtered", 10);
        cluster_pub_ = this->create_publisher<sensor_msgs::msg::PointCloud2>("/hesai/clustered", 10);
        cone_pub_ = this->create_publisher<geometry_msgs::msg::PoseArray>("/hesai/cone_positions", 10);
    }

private:
    void pointCloudCallback(const sensor_msgs::msg::PointCloud2::ConstSharedPtr input)
    {
        // 点云格式转换
        pcl::PointCloud<pcl::PointXYZI>::Ptr cloud(new pcl::PointCloud<pcl::PointXYZI>());
        pcl::fromROSMsg(*input, *cloud);

        auto roi_cloud = extractROI(cloud);
        publishPointCloud(roi_cloud, roi_pub_);

        auto filtered_cloud = groundFilter(roi_cloud);
        publishPointCloud(filtered_cloud, ground_pub_);

        pcl::PointCloud<pcl::PointXYZI>::Ptr clustered_cloud(new pcl::PointCloud<pcl::PointXYZI>());
        //创建一个空的点云对象，并将其指针赋值给 clustered_cloud。clustered_cloud 是一个智能指针，指向一个 pcl::PointCloud<pcl::PointXYZI> 类型的对象。
        //这个点云对象通常用于存储聚类后的点云数据。
        auto cone_positions = clusterAndDetectCones(filtered_cloud, clustered_cloud);
        /*  调用 clusterAndDetectCones 函数对滤波后的点云进行聚类，并检测锥桶。
            聚类结果存储在 clustered_cloud 中。
            锥桶位置信息由 clusterAndDetectCones 返回。*/
        publishPointCloud(clustered_cloud, cluster_pub_);

        publishConePositions(cone_positions);
    }

    pcl::PointCloud<pcl::PointXYZI>::Ptr extractROI(const pcl::PointCloud<pcl::PointXYZI>::Ptr &cloud)
    {

        //ROI区域设置
        float x_min = 0.8, x_max = 40.0;
        float y_min = -3.0, y_max = 3.0;
        float z_min = -1.1, z_max = 0.5;

        pcl::PassThrough<pcl::PointXYZI> pass;//创建一个pass滤波器对象
        pcl::PointCloud<pcl::PointXYZI>::Ptr cloud_filtered(new pcl::PointCloud<pcl::PointXYZI>());//创建一个空的点云对象，并将其指针赋值给 cloud_filtered。

        pass.setInputCloud(cloud);//设置输入点云
        pass.setFilterFieldName("x");//设置滤波字段
        pass.setFilterLimits(x_min, x_max);//设置滤波范围
        pass.filter(*cloud_filtered);//滤波

        pass.setInputCloud(cloud_filtered);
        pass.setFilterFieldName("y");
        pass.setFilterLimits(y_min, y_max);
        pass.filter(*cloud_filtered);

        pass.setInputCloud(cloud_filtered);
        pass.setFilterFieldName("z");
        pass.setFilterLimits(z_min, z_max);
        pass.filter(*cloud_filtered);

        return cloud_filtered;
    }

    pcl::PointCloud<pcl::PointXYZI>::Ptr groundFilter(const pcl::PointCloud<pcl::PointXYZI>::Ptr &cloud)
    {
        pcl::SACSegmentation<pcl::PointXYZI> seg;//创建一个SAC分割器对象
        pcl::PointIndices::Ptr inliers(new pcl::PointIndices());//创建一个点索引对象
        pcl::ModelCoefficients::Ptr coefficients(new pcl::ModelCoefficients());//创建一个模型系数对象
        pcl::PointCloud<pcl::PointXYZI>::Ptr cloud_filtered(new pcl::PointCloud<pcl::PointXYZI>());//创建一个空的点云对象

        seg.setOptimizeCoefficients(true);//设置优化系数
        seg.setModelType(pcl::SACMODEL_PLANE);//设置模型类型为平面
        seg.setMethodType(pcl::SAC_RANSAC);//设置方法类型为RANSAC
        seg.setMaxIterations(90);//设置最大迭代次数
        seg.setDistanceThreshold(0.035);//设置距离阈值

        seg.setInputCloud(cloud);//设置输入点云
        seg.segment(*inliers, *coefficients);

        pcl::ExtractIndices<pcl::PointXYZI> extract;
        extract.setInputCloud(cloud);
        extract.setIndices(inliers);
        extract.setNegative(true);
        extract.filter(*cloud_filtered);

        return cloud_filtered;
    }

    std::vector<Eigen::Vector4f> clusterAndDetectCones(
        const pcl::PointCloud<pcl::PointXYZI>::Ptr &cloud,
        pcl::PointCloud<pcl::PointXYZI>::Ptr &clustered_cloud)
    {
        std::vector<Eigen::Vector4f> cone_positions;
        if (cloud->points.empty()) return cone_positions;//如果输入点云为空，则返回空的锥桶位置信息。（锥桶闪烁）
    
        pcl::search::KdTree<pcl::PointXYZI>::Ptr tree(new pcl::search::KdTree<pcl::PointXYZI>());
        tree->setInputCloud(cloud);//设置输入点云
    
        std::vector<pcl::PointIndices> cluster_indices;
        pcl::EuclideanClusterExtraction<pcl::PointXYZI> ec;//创建一个欧几里德聚类对象
        ec.setClusterTolerance(2.1);  // 调整为更小的聚类容差
        ec.setMinClusterSize(3);      // 适当增加最小点数
        ec.setMaxClusterSize(2000);//最大点数
        ec.setSearchMethod(tree);//设置搜索方法为KD树
        ec.setInputCloud(cloud);//设置输入点云
        ec.extract(cluster_indices);//聚类
    
        for (const auto& indices : cluster_indices) 
        {
            pcl::PointCloud<pcl::PointXYZI>::Ptr cluster(new pcl::PointCloud<pcl::PointXYZI>());
            for (const auto& index : indices.indices)
                cluster->push_back(cloud->points[index]);

                Eigen::Vector4f min_pt, max_pt;//用于存储点云的最小和最大边界点
                pcl::PointCloud<pcl::PointXYZI>::ConstPtr const_cluster = cluster;// 转换为常量指针，不可被修改

            //计算点云的最小和最大边界点
                pcl::getMinMax3D<pcl::PointXYZI>(
                    const_cluster,   
                    "xyz",      
                    -std::numeric_limits<float>::max(),  
                    std::numeric_limits<float>::max(),   
                    min_pt,         
                    max_pt,        
                    false          
                );

            // 计算几何特征
            const float height = max_pt.z() - min_pt.z();
            const float diameter = std::max(
                max_pt.x() - min_pt.x(),
                max_pt.y() - min_pt.y()
            );

            // 锥体过滤条件
            constexpr float MIN_HEIGHT = 0.09f;
            constexpr float MAX_HEIGHT = 0.5f;
            constexpr float MAX_DIAMETER = 0.5f;

            if (height >= MIN_HEIGHT && 
                height <= MAX_HEIGHT &&
                diameter <= MAX_DIAMETER)
            {
                *clustered_cloud += *cluster;

                // 计算质心
                Eigen::Vector4f centroid;
                pcl::compute3DCentroid(*cluster, centroid);
                centroid[3] = static_cast<float>(cluster->size());
                cone_positions.push_back(centroid);
            }
        }
        return cone_positions;
    }


    void publishPointCloud(const pcl::PointCloud<pcl::PointXYZI>::Ptr &cloud,
                          const rclcpp::Publisher<sensor_msgs::msg::PointCloud2>::SharedPtr &publisher)
    {
        sensor_msgs::msg::PointCloud2 output;
        pcl::toROSMsg(*cloud, output);
        output.header.stamp = this->now();
        output.header.frame_id = "hesai_lidar";
        publisher->publish(output);
    }

    void publishConePositions(const std::vector<Eigen::Vector4f> &cone_positions)
    {
        geometry_msgs::msg::PoseArray msg;
        msg.header.stamp = this->now();
        msg.header.frame_id = "hesai_lidar";

        for (const auto &pos : cone_positions)
        {
            geometry_msgs::msg::Pose pose;
            pose.position.x = pos[0];
            pose.position.y = pos[1];
            pose.position.z = pos[2];
            pose.orientation.w = pos[3]; // 存储点云数量
            msg.poses.push_back(pose);
        }
        cone_pub_->publish(msg);
    }

    rclcpp::Subscription<sensor_msgs::msg::PointCloud2>::SharedPtr sub_lidar;
    rclcpp::Publisher<sensor_msgs::msg::PointCloud2>::SharedPtr roi_pub_;
    rclcpp::Publisher<sensor_msgs::msg::PointCloud2>::SharedPtr ground_pub_;
    rclcpp::Publisher<sensor_msgs::msg::PointCloud2>::SharedPtr cluster_pub_;
    rclcpp::Publisher<geometry_msgs::msg::PoseArray>::SharedPtr cone_pub_;
};

int main(int argc, char **argv)
{
    rclcpp::init(argc, argv);
    rclcpp::spin(std::make_shared<HesaiProcessor>());
    rclcpp::shutdown();
    return 0;
}
